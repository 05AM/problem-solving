import java.util.Scanner;

class Main {

	static int size;
	static int[] set;
	static int[] check;
	static String answer = "NO";

	static void dfs(int level) {
		if (level == size) {
			int sum1 = 0;
			int sum2 = 0;

			for (int i = 1; i <= size; i++) {
				if (check[i] == 1) {
					sum1 += set[i];
				} else {
					sum2 += set[i];
				}
			}

			if (sum1 == sum2) {
				answer = "YES";
				return;
			}
		} else {
			check[level + 1] = 1;
			dfs(level + 1);

			check[level + 1] = 0;
			dfs(level + 1);
		}
	}

	public static void main(String[] args) {
		Scanner in = new Scanner(System.in);

		size = in.nextInt();
		set = new int[size + 1];
		check = new int[size + 1];

		for (int i = 1; i <= size; i++) {
			set[i] = in.nextInt();
		}

		Main.dfs(0);
		System.out.println(answer);
	}
}

	static String answer="NO";

		if(flag) return;

			if((total-sum)==sum){

		Main T = new Main();

		for(int i=0; i<n; i++){
			arr[i]=kb.nextInt();
			total+=arr[i];
		}
		T.DFS(0, 0, arr);
		System.out.println(answer);
	}
}
